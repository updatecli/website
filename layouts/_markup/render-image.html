{{- /* Local override of the Thulite images render hook. The Updatecli site references most images either as remote URLs or as absolute paths into static/ (e.g. /images/...), which the upstream hook (built for page/asset resources) cannot process. This version emits a plain responsive-friendly for remote/absolute images and still processes local Hugo resources when they resolve. */ -}} {{- $dest := .Destination -}} {{- $u := urls.Parse $dest -}} {{- $alt := .PlainText -}} {{- $title := .Title -}} {{- $isRemote := or (eq $u.Scheme "http") (eq $u.Scheme "https") -}} {{- $isAbs := strings.HasPrefix $dest "/" -}} {{- if or $isRemote $isAbs -}} {{- else -}} {{- $r := "" -}} {{- with .Page.Resources.GetMatch $dest }}{{ $r = . }}{{ end -}} {{- if not $r }}{{ with resources.Get $dest }}{{ $r = . }}{{ end }}{{ end -}} {{- if $r -}} {{- else -}} {{- end -}} {{- end -}}
